import shlex
import subprocess

show = subprocess.run(
    ["systemctl", "show", "burn-p2p-bootstrap", "--no-pager", "-p", "EnvironmentFiles"],
    check=False,
    text=True,
    capture_output=True,
)
line = show.stdout.strip()
value = line.split("=", 1)[1] if "=" in line else ""
paths = []
for item in shlex.split(value):
    path = item.split(":", 1)[0]
    if path and path != "-":
        paths.append(path)

if not paths:
    print("no environment files configured")
for path in paths:
    print(f"env_file={path}")
    try:
        lines = open(path, encoding="utf-8").read().splitlines()
    except OSError as error:
        print(f"  unreadable: {error}")
        continue
    for raw in lines:
        stripped = raw.strip()
        if not stripped or stripped.startswith("#"):
            continue
        key = stripped.split("=", 1)[0].strip()
        print(f"  {key}=<redacted>")
PY

from pathlib import Path
import json

path = Path("/etc/burn-dragon-p2p/bootstrap.json")
if not path.exists():
    print(f"missing {path}")
    raise SystemExit(0)

config = json.loads(path.read_text())
for label, root in [
    ("spec", config.get("spec", {})),
    ("bootstrap_peer.node", config.get("bootstrap_peer", {}).get("node", {})),
]:

        for value in root.get(key, []) or []:
            print(f"  {key}: {value}")
